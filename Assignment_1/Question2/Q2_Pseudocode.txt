START
PRINT "Coordinates are represented in the manner (x,y)"
PRINT "Enter the x-coordinate: "
INPUT x
PRINT "Enter the y-coordinate: "
INPUT y
IF x > 0 AND y > 0 THEN
    PRINT "Point lies in Quadrant I"
ELSEIF x > 0 AND y < 0 THEN
    PRINT "Point lies in Quadrant IV"
ELSEIF x < 0 AND y > 0 THEN
    PRINT "Point lies in Quadrant II"
ELSEIF x < 0 AND y < 0 THEN
    PRINT "Point lies in Quadrant III"
ELSE
    PRINT "Point lies on Axes"
ENDIF
END
